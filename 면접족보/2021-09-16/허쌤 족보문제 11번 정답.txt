public class PrimeNumber {

	public static void main(String[] args) {	
		
		for(int i =2; i<=100; i++)	{
			
			if(isPrimeNumber(i)) {
				System.out.println(i);
			}
			
		}
	}
		
	public static boolean isPrimeNumber(int k) {
		for(int i = 2; i< k; i++) {
			if( k % i == 0) 
				return false;
		}
		return true;
	}
}
